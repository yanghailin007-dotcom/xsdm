```python
def decompose_major_event_with_density_constraints(major_event):

    required_loots = major_event.core_selling_points_contribution.required_frequency.loot_count
```

#### B. 实时进度检查机制

```python
class SellingPointDensityMonitor:
    """
    卖点密度监控器（生成时实时检查）
    """
    
    def __init__(self, stage_plan):
        self.stage_plan = stage_plan
        self.loot_history = []  # 记录每章是否掠夺
        self.combat_chapters = []  # 记录战斗章节
    
    def pre_generation_check(self, chapter_num):
        """
        生成前检查：返回必须执行的约束
        """
        constraints = []
        
        # 检查1：掠夺间隔
        chapters_since_loot = self.get_chapters_since_last_loot(chapter_num)
        if chapters_since_loot >= 3:
            constraints.append({
                "type": "must_loot",
                "urgency": "critical",
                "message": f"已{chapters_since_loot}章未掠夺，本章必须掠夺！"
            })
        
        # 检查2：连续战斗
        consecutive_combat = self.get_consecutive_combat_chapters(chapter_num)
        if consecutive_combat >= 3:
            constraints.append({
                "type": "must_change_pace",
                "urgency": "high",
                "message": f"已连续{consecutive_combat}章战斗，必须切换节奏！"
            })
        
        # 检查3：10章密度
        loot_count_10 = self.get_loot_count_in_range(chapter_num-10, chapter_num)
        if loot_count_10 < 3:
            constraints.append({
                "type": "must_meet_density",
                "urgency": "high",
                "message": f"最近10章仅掠夺{loot_count_10}次，密度不足！"
            })
        
        return constraints
    
    def post_generation_validation(self, chapter_num, generated_content):
        """
        生成后验证：检查是否符合约束
        """
        validation_result = {
            "passed": True,
            "violations": [],
            "suggestions": []
        }
        
        # 提取本章的卖点信息
        chapter_loots = extract_loot_info(generated_content)
        chapter_type = classify_chapter_type(generated_content)
        
        # 验证1：是否满足掠夺要求
        constraints = self.pre_generation_check(chapter_num)
        for constraint in constraints:
            if constraint["type"] == "must_loot" and len(chapter_loots) == 0:
                validation_result["passed"] = False
                validation_result["violations"].append(
                    f"违反约束：{constraint['message']}"
                )
                validation_result["suggestions"].append(
                    "建议：添加掠夺情节，或在下一章开头立即补上"
                )
        
        # 验证2：是否重复场景
        if chapter_type == "combat" and self.combat_chapters[-3:] == ["combat"] * 3:
            validation_result["passed"] = False
            validation_result["violations"].append(
                "连续4章战斗场景，违反多样性规则"
            )
        
        # 记录历史
        self.loot_history.append({
            "chapter": chapter_num,
            "loots": chapter_loots,
            "type": chapter_type
        })
        
        return validation_result
```

```python
def auto_fix_density_issues(chapter_num, validation_result):
    """
    当密度检查失败时，自动修正下一章的生成策略
    """
    if not validation_result["passed"]:
        fixes = []
        
        for violation in validation_result["violations"]:
            if "掠夺" in violation:
                fixes.append({
                    "action": "force_loot_next_chapter",
                    "params": {
                        "chapter": chapter_num + 1,
                        "must_include": ["目标锁定", "击杀", "掠夺", "词条展示"],
                        "target_options": get_available_high_value_targets(chapter_num + 1)
                    }
                })
            
            if "连续战斗" in violation:
                fixes.append({
                    "action": "change_pace_next_chapter",
                    "params": {
                        "chapter": chapter_num + 1,
                        "force_type": "plot_advancement_or_loot_result",
                        "avoid": "pure_combat"
                    }
                })
        
        return fixes
```

```python
def batch_validate_selling_point_density(chapters, stage_plan):
    """
    批量验证卖点密度
    """
    report = {
        "overall_score": 0,
        "issues": [],
        "recommendations": []
    }
    
    # 检查1：每个major_event的掠夺密度
    for major_event in stage_plan.major_events:
        event_chapters = get_chapters_in_range(chapters, major_event.chapter_range)
        loot_count = count_loots(event_chapters)
        expected = major_event.core_selling_points_contribution.required_frequency.loot_count
        
        if loot_count < expected:
            report["issues"].append({
                "severity": "high",
                "event": major_event.name,
                "problem": f"掠夺不足：期望{expected}次，实际{loot_count}次",
                "chapters": [c.chapter_number for c in event_chapters if not c.has_loot]
            })
    
    # 检查2：最大无掠夺间隔
    max_gap = find_max_gap_without_loot(chapters)
    if max_gap > 3:
        report["issues"].append({
            "severity": "critical",
            "problem": f"存在{max_gap}章连续无掠夺的区间",
            "recommendation": "需要在这些章节中插入掠夺情节"
        })
    
    # 检查3：卖点类型多样性
    loot_types = classify_loots_by_type(chapters)
    if len(loot_types) < 3:
        report["issues"].append({
            "severity": "medium",
            "problem": f"掠夺类型单一：仅有{list(loot_types.keys())}",
            "recommendation": "需要增加其他类型的词条掠夺"
        })
    
    # 计算总分
    report["overall_score"] = calculate_density_score(report["issues"])
    
    return report

def calculate_density_score(issues):
    """
    根据问题严重程度计算密度得分（0-10）
    """
    score = 10
    for issue in issues:
        if issue["severity"] == "critical":
            score -= 3
        elif issue["severity"] == "high":
            score -= 2
        elif issue["severity"] == "medium":
            score -= 1
    return max(0, score)
```

```python
class SystemPowerDensityMonitor(SellingPointDensityMonitor):
```
