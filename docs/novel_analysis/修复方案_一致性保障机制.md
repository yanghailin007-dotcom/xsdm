```python
def _validate_chapter_content_consistency(self, chapter_data: Dict, chapter_number: int, novel_title: str) -> List[Dict]:
    """验证章节内容的一致性"""
    issues = []
    content = chapter_data.get("content", "")
    
    # 1. 加载角色发展数据
    character_file = os.path.join(self.quality_assessor.storage_path, f"{novel_title}_character_development.json")
    if not os.path.exists(character_file):
        return issues
    
    with open(character_file, 'r', encoding='utf-8') as f:
        characters = json.load(f)
    
    # 2. 检查死亡角色是否出现在内容中
    for char_name, char_data in characters.items():
        status = char_data.get('status', 'active')
        if status in ['dead', '死亡', '阵亡', '牺牲']:
            # 检查这个死亡角色是否在内容中出现
            if char_name in content:
                # 进一步检查是否是以回忆、幻象等合理形式出现
                if not self._is_ghost_appearance_valid(content, char_name):
                    issues.append({
                        'type': 'DEAD_CHARACTER_APPEARED',
                        'severity': '高',
                        'description': f"已死亡角色 {char_name} 在第{chapter_number}章中出现",
                        'suggestion': "删除该角色的出场或改为回忆/幻象形式"
                    })
    
    # 3. 检查状态矛盾
    for char_name, char_data in characters.items():
        if char_name not in content:
            continue
        
        # 检查内容中是否有矛盾描述
        status = char_data.get('status', 'active')
        location = char_data.get('attributes', {}).get('location', '')
        
        if status == 'dead':
            # 死亡角色不应该有行动描述
            action_patterns = [
                f"{char_name}站起身",
                f"{char_name}走过去",
                f"{char_name}说:",
                f"{char_name}道:"
            ]
            for pattern in action_patterns:
                if pattern in content:
                    issues.append({
                        'type': 'DEAD_CHARACTER_ACTION',
                        'severity': '高',
                        'description': f"已死亡角色 {char_name} 在内容中有行动描述",
                        'suggestion': "删除该行动或改为回忆"
                    })
    
    return issues

def _is_ghost_appearance_valid(self, content: str, char_name: str) -> bool:
    """检查死亡角色的出现是否合理(如回忆、幻象)"""
    # 检查上下文关键词
    valid_contexts = [
        "回忆",
        "想起",
        "梦中",
        "幻觉",
        "幻象",
        "仿佛",
        "宛如",
        "往事"
    ]
    
    # 提取包含角色名的句子
    import re
    sentences = re.split(r'[。！？\n]', content)
    
    for sentence in sentences:
        if char_name in sentence:
            # 检查是否有合理上下文
            if any(ctx in sentence for ctx in valid_contexts):
                return True
    
    return False
```

```python
def _validate_existing_scenes(self, scenes: List[Dict], chapter_number: int) -> bool:
    """验证现有场景是否完整可用"""
    if not scenes:
        return False
    
    # 检查场景数量是否合理
    if len(scenes) < 3:
        self.logger.warn(f"  现有场景数量过少: {len(scenes)}")
        return False
    
    # 检查必需的场景类型
    scene_positions = {scene.get('position') for scene in scenes}
    required_positions = {'opening', 'climax', 'ending'}
    
    missing_positions = required_positions - scene_positions
    if missing_positions:
        self.logger.warn(f"  缺少必需场景: {missing_positions}")
        return False
    
    # 检查每个场景的必需字段
    for scene in scenes:
        if not scene.get('name') or not scene.get('purpose'):
            self.logger.warn(f"  场景缺少必需字段: {scene}")
            return False
    
    return True

def _persist_scene_events(self, chapter_number: int, scenes: List[Dict], plan_container: Dict):
    """持久化场景事件"""
    stage_name = plan_container.get('stage_name', '未知阶段')
    
    # 确保event_system和chapter_scene_events存在
    if 'event_system' not in plan_container:
        plan_container['event_system'] = {}
    if 'chapter_scene_events' not in plan_container['event_system']:
        plan_container['event_system']['chapter_scene_events'] = []
    
    # 查找或创建章节条目
    chapter_entry = None
    for entry in plan_container['event_system']['chapter_scene_events']:
        if entry.get('chapter_number') == chapter_number:
            chapter_entry = entry
            break
    
    if chapter_entry:
        chapter_entry['scene_events'] = scenes
    else:
        chapter_entry = {
            'chapter_number': chapter_number,
            'chapter_goal': scenes[0].get('purpose', '生成章节内容') if scenes else '',
            'writing_focus': '',
            'scene_events': scenes
        }
        plan_container['event_system']['chapter_scene_events'].append(chapter_entry)
    
    # 保存到文件
    self.novel_generator.stage_plan_manager.save_and_cache_stage_plan(
        stage_name=stage_name,
        plan_data=plan_container
    )
```

```python
def _is_too_similar_to_previous(self, new_chapter: Dict, novel_data: Dict, chapter_number: int) -> bool:
    """检查新章节是否与之前的尝试过于相似"""
    # 获取之前尝试的内容
    previous_attempts = novel_data.get('_generation_attempts', {}).get(str(chapter_number), [])
    
    if not previous_attempts:
        return False
    
    new_content = new_chapter.get("content", "")
    
    for prev_content in previous_attempts:
        # 计算相似度
        similarity = self._calculate_text_similarity(new_content, prev_content)
        if similarity > 0.85:  # 85%相似度阈值
            return True
    
    return False

def _calculate_text_similarity(self, text1: str, text2: str) -> float:
    """计算两个文本的相似度"""
    from difflib import SequenceMatcher
    
    # 简单的序列匹配
    return SequenceMatcher(None, text1, text2).ratio()
```

```python
def _cleanup_generated_chapters(self, novel_data: Dict, current_chapter: int) -> Dict:
    """改进的章节清理策略"""
    if "generated_chapters" not in novel_data:
        return novel_data
    
    generated_chapters = novel_data["generated_chapters"]
    if not isinstance(generated_chapters, dict):
        return novel_data
    
    # ✅ 改进: 保留最近5章 + 重要转折章节
    recent_chapters_to_keep = 5
    chapters_to_process = list(generated_chapters.keys())
    chapters_to_process.sort()
    
    # 识别重要转折章节(包含死亡、重大事件等)
    milestone_chapters = self._identify_milestone_chapters(generated_chapters)
    
    cleaned_chapters = {}
    
    # 1. 保留最近的章节
    for chapter_num in reversed(chapters_to_process):
        if len(cleaned_chapters) < recent_chapters_to_keep:
            chapter_data = generated_chapters[chapter_num]
            cleaned_chapters[chapter_num] = chapter_data
    
    # 2. 保留里程碑章节
    for milestone_chapter in milestone_chapters:
        if milestone_chapter not in cleaned_chapters:
            cleaned_chapters[milestone_chapter] = generated_chapters[milestone_chapter]
    
    # 3. 其他章节只保留关键信息
    for chapter_num in chapters_to_process:
        if chapter_num not in cleaned_chapters:
            chapter_data = generated_chapters[chapter_num]
            cleaned_chapters[chapter_num] = self._create_chapter_summary(chapter_data)
    
    novel_data["generated_chapters"] = cleaned_chapters
    return novel_data

def _identify_milestone_chapters(self, chapters: Dict) -> List[int]:
    """识别重要的转折章节"""
    milestones = []
    
    for chapter_num, chapter_data in chapters.items():
        # 检查章节标记
        if chapter_data.get('is_milestone', False):
            milestones.append(chapter_num)
            continue
        
        # 检查内容中的死亡事件
        content = chapter_data.get('content', '')
        death_keywords = ['死亡', '陨落', '牺牲', '阵亡', '被杀']
        if any(keyword in content for keyword in death_keywords):
            milestones.append(chapter_num)
            continue
        
        # 检查质量评估中的标记
        quality_data = chapter_data.get('quality_assessment', {})
        if quality_data.get('contains_major_plot_change'):
            milestones.append(chapter_num)
    
    return milestones

def _create_chapter_summary(self, chapter_data: Dict) -> Dict:
    """创建章节摘要"""
    return {
        "chapter_title": chapter_data.get("chapter_title", ""),
        "word_count": chapter_data.get("word_count", 0),
        "quality_score": chapter_data.get("quality_score", 0),
        "next_chapter_hook": chapter_data.get("next_chapter_hook", ""),
        "connection_to_previous": chapter_data.get("connection_to_previous", ""),
        "key_events": chapter_data.get("key_events", []),
        "character_deaths": chapter_data.get("character_deaths", []),
        "is_summary": True
    }
```

## 总结

通过以上六个修复方案,可以系统性地解决"重复写"和"人物死了又活"的问题:

1. **统一状态管理** - 消除双轨制导致的不一致
2. **增强一致性检查** - 在生成后验证内容
3. **强化Prompt指令** - 明确禁止不合理内容
4. **改进重试机制** - 避免重复生成相似内容
5. **状态同步** - 确保两个状态系统一致
6. **优化上下文管理** - 保留重要信息

建议按优先级逐步实施,每次实施后进行充分测试验证。
